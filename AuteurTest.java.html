package Bib;
import static org.junit.Assert.assertEquals;
import static org.junit.jupiter.api.Assertions.*;

import org.junit.jupiter.api.Test;

class AuteurTest {
	@Test
	void testAuteur() {
		String value = "ayoub";
	    Auteur m = new Auteur(value);
	    String result = m.getNomEtr();
	    assertEquals(value, result);
	}

	@Test
	void testToString() {
		assertFalse(new Auteur(null).toString().contains("ayoub"));
	}

	@Test
	void testGetNomEtr() {
		 String longValue = "aidi";
		 Auteur m = new Auteur(longValue);
		    assertEquals(longValue, m.getNomEtr());
	}

	@Test
	void testSetNomEtr() {
		Auteur m = new Auteur("ayoub");
	      m.setNomEtr("elaidi");
	      assertSame("elaidi", m.getNomEtr());
	}

}
