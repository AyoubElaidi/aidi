package Bib;

import static org.junit.Assert.assertEquals;

import static org.junit.jupiter.api.Assertions.*;

class LivreTest {

	@Test
	void testLivre() {
		String value = "elaidi";
	    Livre m = new Livre(value);
	    String result = m.getTitre();
	    assertEquals(value, result);
	}
	
	@Test
	void testGetTitre() {
		 String longValue = "aidi";
	    
	    Livre m = new Livre(longValue);
	    assertEquals(longValue, m.getTitre());
	}

	void testToString()
	{
		assertFalse(new Livre(null).toString().contains("ayoub"));
	}
	@Test
	void testSetTitre() {
		  Livre m = new Livre("ayoub");
	      m.setTitre("elaidi");

	}
